# 视频导入与智能章节添加功能实现计划

## 1. 需求概述

### 1.1 目标
为视频进度条生成器添加视频导入和智能章节添加功能，用户可以通过导入视频、拖动播放轴到指定位置、一键添加章节。

### 1.2 用户决策（已确认）

| 决策项 | 选择 |
|--------|------|
| 布局位置 | 替换右侧预览区 |
| 章节添加方式 | 仅手动添加（拖动播放轴 + 点击添加） |
| 缩略图预览 | 不需要 |
| 视频播放器 | Video.js |
| 自动填时长 | 是，导入视频后自动填入总时长 |

#### 3.2 Video.js 初始化函数
```javascript
function initVideoPlayer() {
    // 动态加载 Video.js（如果未加载）
    if (!window.videojs) {
        loadVideoJS().then(() => createVideoPlayer());
    } else {
        createVideoPlayer();
    }
}

function createVideoPlayer() {
    const videoEl = document.getElementById('videoPlayer');
    videoPlayer = videojs(videoEl, {
        controls: true,
        autoplay: false,
        preload: 'auto',
        fluid: true,
        responsive: true,
        playbackRates: [0.5, 1, 1.5, 2],
        controlBar: {
            children: [
                'playToggle',
                'volumePanel',
                'currentTimeDisplay',
                'timeDivider',
                'durationDisplay',
                'progressControl',
                'playbackRateMenuButton',
                'fullscreenToggle'
            ]
        }
    });

    // 监听时间更新
    videoPlayer.on('timeupdate', updateVideoTimeDisplay);

    // 视频加载完成后自动填入时长
    videoPlayer.on('loadedmetadata', () => {
        const duration = videoPlayer.duration();
        if (duration > 0) {
            document.getElementById('totalDuration').value = formatTime(duration);
            schedulePreview();
        }
    });
}
```

#### 3.3 视频时间显示更新
```javascript
function updateVideoTimeDisplay() {
    if (!videoPlayer) return;
    const current = videoPlayer.currentTime() || 0;
    const duration = videoPlayer.duration() || 0;
    document.getElementById('videoTimeDisplay').textContent =
        `${formatTime(current)} / ${formatTime(duration)}`;
}
```

#### 3.4 视频导入处理
```javascript
async function handleVideoImport(file) {
    if (!file) return;

    // 验证文件类型
    const validTypes = ['video/mp4', 'video/webm', 'video/quicktime'];
    if (!validTypes.includes(file.type)) {
        setStatus('不支持的视频格式，请使用 MP4、WebM 或 MOV', 'error');
        return;
    }

    // 限制文件大小（500MB）
    const maxSize = 500 * 1024 * 1024;
    if (file.size > maxSize) {
        setStatus('视频文件过大，请使用 500MB 以下的文件', 'error');
        return;
    }

    currentVideoFile = file;
    const url = URL.createObjectURL(file);

    if (!videoPlayer) {
        initVideoPlayer();
    }

    // 重置并加载视频
    videoPlayer.src({ src: url, type: file.type });
    videoPlayer.load();

    // 显示视频容器，隐藏图片容器
    document.getElementById('videoContainer').hidden = false;
    document.getElementById('imageContainer').hidden = true;

    setStatus('视频已导入，拖动播放轴到指定位置后点击添加章节', 'ok');
}
```

#### 3.5 添加当前时间为章节
```javascript
function addChapterAtCurrentTime() {
    if (!videoPlayer) {
        setStatus('请先导入视频', 'error');
        return;
    }

    const currentTime = videoPlayer.currentTime();
    if (!currentTime || currentTime <= 0) {
        setStatus('视频尚未开始播放', 'error');
        return;
    }

    const formattedTime = formatTime(currentTime);
    chapters.push({
        time: formattedTime,
        title: '新章节'
    });

    // 排序并重新渲染
    sortChaptersByTime();
    renderChapters();
    saveState();
    schedulePreview();

    setStatus(`已添加章节：${formattedTime}`, 'ok');
}

function sortChaptersByTime() {
    chapters.sort((a, b) => {
        return parseTimeToSeconds(a.time) - parseTimeToSeconds(b.time);
    });
}

function parseTimeToSeconds(timeStr) {
    const parts = timeStr.split(':').map(Number);
    if (parts.length === 1) return parts[0];
    if (parts.length === 2) return parts[0] * 60 + parts[1];
    if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
    return 0;
}
```

#### 3.6 清空视频
```javascript
function clearVideo() {
    if (videoPlayer) {
        videoPlayer.pause();
        videoPlayer.src('');
        videoPlayer.load();
    }
    currentVideoFile = null;

    document.getElementById('videoContainer').hidden = true;
    document.getElementById('imageContainer').hidden = false;
}
```

### Step 5: 后端视频时长获取（可选增强）

**文件**: `app.py`

如需后端获取视频时长（用于校验），添加接口:
```python
@app.route("/video_metadata", methods=["POST"])
def video_metadata():
    if "video" not in request.files:
        return jsonify({"success": False, "message": "请上传视频文件"}), 400

    video_file = request.files["video"]

    # 使用 imageio-ffmpeg 获取时长
    import tempfile
    import imageio_ffmpeg

    with tempfile.NamedTemporaryFile(delete=False, suffix=".mp4") as tmp:
        video_file.save(tmp.name)
        probe = imageio_ffmpeg.get_ffmpeg_exe()
        # 调用 ffprobe 获取时长
        result = subprocess.run(
            [probe.replace("ffmpeg", "ffprobe"), "-v", "error",
             "-show_entries", "format=duration", "-of",
             "default=noprint_wrappers=1:nokey=1", tmp.name],
            capture_output=True, text=True
        )
        duration = float(result.stdout.strip()) if result.stdout else 0
        os.unlink(tmp.name)

    return jsonify({
        "success": True,
        "duration": duration,
        "formatted": format_time(duration)
    })
```
